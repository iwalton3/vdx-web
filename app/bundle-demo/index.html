Quick Start

<script type="module">
import { defineComponent, html } from './framework-bundle.js';

defineComponent('my-counter', {
    data() {
        return { count: 0 };
    },

    methods: {
        increment() { this.state.count++; }
    },

    template() {
        return html`
            <div>
                <h2>Count: ${this.state.count}</h2>
                <button on-click="increment">+1</button>
            </div>
        `;
    }
});
</script>
